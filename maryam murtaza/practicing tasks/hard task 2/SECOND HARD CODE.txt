LONGEST VALID PARANTHESES

class Solution(object):
    def longestValidParentheses(self, s):
        stack = [-1]
        max_length = 0

        for i in range(len(s)):
            if s[i] == '(':
                stack.append(i)
            else:
                stack.pop()
                if len(stack) == 0:
                    stack.append(i)
                else:
                    length = i - stack[-1]
                    max_length = max(max_length, length)

        return max_length


input_string = "(()))())"
solution = Solution()
result = solution.longestValidParentheses(input_string)
print(result)
